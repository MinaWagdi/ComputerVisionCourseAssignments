def addMultNoise(original_im, standard_dev):
    #GaussRand=np.random.normal(loc=0,scale=standard_dev,size=None)
    R=original_im.shape[0]
    C=original_im.shape[1]
    #newIm=np.zeros((R,C),original_im.dtype)
    for i in range(R):
        for j in range(C):
            GaussRand=np.random.normal(loc=0,scale=standard_dev,size=None)
            original_im[i,j]=multiplicativeNoiseFunction(GaussRand,original_im[i][j])
    return original_im

def calcTotalLocalVariance(Im):
    R=Im.shape[0]
    print ('R : ', R )
    C=Im.shape[1]
    print ('C : ', C )
    # i and j are the center of the window
    # we subtacted 2 from R and C in the for loop because the last index is R-1 and C-1 and the center 
    # will be the index before it
    TotalLocalVariance = 0
    for i in range(1,R-1):
        for j in range(1,C-1):
            windowAverage=calcWindowAvg(Im,i,j)
            #print('i :', i)
            #print('j :', j)
            #print('windowAverage : ' , windowAverage)
            temp=0
            for k in range(i-1,i+2):
                for l in range(j-1,j+2):
                    temp= temp+((Im[k][l]-windowAverage)**2)
            TotalLocalVariance+=(temp/9)
    return TotalLocalVariance

def increaseContrast(originalImage,A,B,C,D):
    r=originalImage.shape[0]
    c=originalImage.shape[1]
    newImage=originalImage.copy()
    for i in range(0,r-1):
        for j in range(0,c-1):
            if originalImage[i][j] in range(0,A) : 
                newImage[i][j]=(B/A)*originalImage[i][j]
            elif originalImage[i][j]in range(A,C) : 
                slope=(D-B)/(C-A)
                b = B-(A*slope)
                newImage[i][j]=slope*originalImage[i][j]+b
            elif originalImage[i][j]in range(C,255):
                slope=(255-D)/(255-C)
                b = D-slope*C
                newImage[i][j]=originalImage[i][j]*slope+b
    return newImage
